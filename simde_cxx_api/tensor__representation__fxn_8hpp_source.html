69 template<typename... Args>
70 auto tensor_representation(pluginplay::Module& mod, const Args&... args) {
71  constexpr auto n_add_args = sizeof...(Args); // # of additional arguments
72  constexpr auto n_center = n_add_args - 1; // One arg is an operator
73 
74  // Basic compile-time error checking
75  static_assert(n_center >= 2, "Minimally must be a two-center integral");
76  static_assert(n_center <= 4, "Only support up to four-center integrals");
77 
78  auto&& [op, bases] = detail_::split_basis_op(args...);
79 
80  detail_::TensorRepParser p(bases);
81 
82  using transform_traits_t = detail_::TransformTraits<std::decay_t<Args>...>;
83  constexpr bool has_tots = transform_traits_t::has_tots;
84  constexpr bool has_derived = transform_traits_t::has_derived;
85  constexpr bool is_general = has_tots;
86  constexpr bool is_derived = has_derived && (!is_general);
87  constexpr bool is_all_ao = (!has_derived) && (!is_general);
88 
89  using op_type = std::decay_t<decltype(op)>;
90 
91  if constexpr(is_all_ao) {
92  return detail_::ao_dispatch<n_center, op_type>(mod, p.m_ao_spaces, op);
93  } else if constexpr(is_derived) {
94  using pt = TransformedTensorRepresentation<n_center, op_type>;
95  // Some of the spaces may be independent, need to handle this
96 
97  return mod.run_as<pt>(p.m_ao_spaces, p.m_derived_spaces, op);
98  } else {
99  using pt = GeneralTransformedTensorRepresentation<n_center, op_type>;
100  return mod.run_as<pt>(p.m_ao_spaces, p.m_derived_spaces, p.m_tot_spaces,
101  // p.m_ind_spaces,
102  op);
103  }
104 }
