41 template<typename... Args>
42 auto split_basis_op(const Args&... args) {
43  constexpr auto n_args = sizeof...(Args);
44  auto arg_tuple = std::tie(args...);
45 
46  // This could be done with TMP using some fancy traits to grab only select
47  // elements out of a tuple, but those traits don't exist in utilities at the
48  // moment.
49  if constexpr(n_args == 3) {
50  auto&& op = std::get<1>(arg_tuple);
51  using op_type = std::decay_t<decltype(op)>;
52  constexpr auto is_op = chemist::operators::is_operator_v<op_type>;
53  static_assert(is_op, "For two centers, Op must be argument 1");
54 
55  auto&& bra = std::get<0>(arg_tuple);
56  auto&& ket = std::get<2>(arg_tuple);
57  using tuple_type =
58  std::tuple<decltype(std::as_const(op)), decltype(std::tie(bra, ket))>;
59  return tuple_type(std::as_const(op), std::tie(bra, ket));
60  } else if constexpr(n_args == 4) {
61  auto&& op = std::get<1>(arg_tuple);
62  using op_type = std::decay_t<decltype(op)>;
63  constexpr auto is_op = chemist::operators::is_operator_v<op_type>;
64  static_assert(is_op, "For three centers, Op must be argument 1");
65 
66  auto&& bra = std::get<0>(arg_tuple);
67  auto&& ket1 = std::get<2>(arg_tuple);
68  auto&& ket2 = std::get<3>(arg_tuple);
69  using tuple_type = std::tuple<decltype(std::as_const(op)),
70  decltype(std::tie(bra, ket1, ket2))>;
71  return tuple_type(std::as_const(op), std::tie(bra, ket1, ket2));
72  } else if constexpr(n_args == 5) {
73  auto&& op = std::get<2>(arg_tuple);
74  using op_type = std::decay_t<decltype(op)>;
75  constexpr auto is_op = chemist::operators::is_operator_v<op_type>;
76  static_assert(is_op, "For four centers, Op must be argument 2");
77 
78  auto&& bra1 = std::get<0>(arg_tuple);
79  auto&& bra2 = std::get<1>(arg_tuple);
80  auto&& ket1 = std::get<3>(arg_tuple);
81  auto&& ket2 = std::get<4>(arg_tuple);
82 
83  using tuple_type =
84  std::tuple<decltype(std::as_const(op)),
85  decltype(std::tie(bra1, bra2, ket1, ket2))>;
86  return tuple_type(std::as_const(op), std::tie(bra1, bra2, ket1, ket2));
87  } else {
88  // N.B. If we get here we know n_args != 3, 4, or 5
89  static_assert(n_args == 3,
90  "Only support 2, 3, or 4-centered integrals");
91  }
92 }
